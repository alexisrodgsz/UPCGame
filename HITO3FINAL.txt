	for (f = 0; f < filas; f++)
	{
		for (c = 0; c < columnas; c++)
		{
			if (nivel == 1) { matriz_general[f][c] = mapa_juego1[f][c]; }
		}

void mostrar_mapa() {
	int f, c;
	Console::BackgroundColor = ConsoleColor::Blue;
	for (f = 0; f < filas; f++)
	{
		for (c = 0; c < columnas; c++)
		{
			if (matriz_general[f][c] == 0) { cout << " "; }
			if (matriz_general[f][c] == 1) { Console::ForegroundColor = ConsoleColor::DarkGreen; cout << (char)178; }
			if (matriz_general[f][c] == 2) { Console::ForegroundColor = ConsoleColor::Black; cout << (char)219; }
			if (matriz_general[f][c] == 3) { Console::ForegroundColor = ConsoleColor::DarkBlue; cout << (char)219; }
			if (matriz_general[f][c] == 4) { Console::ForegroundColor = ConsoleColor::DarkRed; cout << (char)219; }
			if (matriz_general[f][c] == 6) { Console::ForegroundColor = ConsoleColor::Red; cout << (char)219; }
			if (matriz_general[f][c] == 7) { Console::ForegroundColor = ConsoleColor::White; cout << (char)219; }
		}
		cout << endl;
	}
}

void Movpez(int &Pex1, int &Pey1, int &Change)
{

		DibujaPescadoIzquierdo(Pex1, Pey1);

	if (Change == 1)

void Movimiento()
{
	int X = 6, Y = 6; //posicion de la caña
	int tx1 = 10, ty1 = 13;//tiburon
	int t1 = 1;
	int Pex1 = 5, Pey1 = 17, Change = 1;
	int Pex2 = 36, Pey2 = 19, Change2 = -1;
	int Pex3 = 30, Pey3 = 21, Change3 = 1;
	int Pex4 = 26, Pey4 = 23, Change4 = -1;
	int Pex5 = 29, Pey5 = 25, Change5 = 1;
	int Pex6 = 31, Pey6 = 27, Change6 = -1;
	char Tecla;
	int X1 = X;
	int Y1 = Y;
	int x1, y1, x2, y2; // caña
	x1 = X; y1 = Y + 1; x2 = X; y2 = Y;
	int Point = 0;
	DibujaCaña(X, Y);
	puntaje(Point);
	while (1)
	{
		while (!kbhit())
		{
			if ((X >= Pex1) && (Y >= Pey1) && (X <= Pex1 + 4) && (Y <= (Pey1 + 2)) && (Change == 1))
				Change = 0;
			if (Change == 0)
			{
				Pex1 = X; Pey1 = Y;
				if (Y == 6) // suma puntos en linea 
				{
					 Pex1 = 5; Pey1 = 17; Change = 1; // generar pez en su mismo lugar
					Point = Point + 3; // contador de puntos
					puntaje(Point);

				}
				if ((X >= tx1) && (Y >= ty1) && (X <= tx1 + 4) && (Y <= (ty1 + 2)) && (Change == 0))
				{
					 Pex1 = 5; Pey1 = 17; Change = 1; //generar pez en su mismo lugar
				}
			}
	
			if ((X >= Pex3) && (Y >= Pey3) && (X <= Pex3 + 4) && (Y <= (Pey3 + 2)) && (Change3 == 1))
				Change3 = 0;
			if (Change3 == 0)
			{
				Pex3 = X; Pey3 = Y;
				if (Y == 6) // suma puntos en linea 3
				{
				    Pex3 =30; Pey3 = 21; Change3 = 1;
					Point = Point + 3; // contador de puntos
					puntaje(Point);
		
				}
				if ((X >= tx1) && (Y >= ty1) && (X <= tx1 + 4) && (Y <= (ty1 + 2)) && (Change3 == 0))
				{
					 Pex3 = 30; Pey3 = 21; Change3 = 1;
				}
			}
			Movtibu(tx1, ty1, t1);  // tiburon
			Movpez(Pex1, Pey1, Change );
			Movpez(Pex2, Pey2, Change2 );
			Movpez(Pex3, Pey3, Change3 );
			Movpez(Pex4, Pey4, Change4 );
			Movpez(Pex5, Pey5, Change5 );
			Movpez(Pex6, Pey6, Change6 );
			Cañapescar(x1, y1, x2, y2);
			_sleep(30);
			// Borra Peces
			BorraPez(Pex1, Pey1);
			BorraPez(Pex2, Pey2);
			BorraPez(Pex3, Pey3);
			BorraPez(Pex4, Pey4);
			BorraPez(Pex5, Pey5);
			BorraPez(Pex6, Pey6);
			BorraTiburon(tx1, ty1);
		}

		Tecla = getch();

		//parte vertical de la caña
		if (Tecla == 72) //arriba
		{
			BorraCañapescar(x1, y1, x2, y2);
			Y--;
			if (Y == 0)  //si llega al limite arriba regresa una posicion
				Y++;
			y2 = Y;
		}
		if (Tecla == 80) //abajo
		{
			Y++;
			if (Y == 24) //si llega al limite infrior regresa una posicion
				Y--;
			y2 = Y;
		}
		//parte horizontal de la caña
		if (Tecla == 75) //izquierda
		{
			BorraCaña(X1, Y1); //borra linea horizontal
			BorraCañapescar(x1, y1, x2, y2); //borra linera vertical
			X--;
			if (X == 7) //si llega al limite izquirdo regresa una posicion
				X++;
			x1 = X;  
			x2 = X;
			X1 = X; 
		}
		if (Tecla == 77)  //derecha
		{
			BorraCañapescar(x1, y1, x2, y2);
			X++;
			if (X == 39) //si llega al limite derecho regresa una posicion
				X--;
			x1 = X;
			x2 = X;
			X1 = X;
		}
		DibujaCaña(X1, Y1);

	}
}
